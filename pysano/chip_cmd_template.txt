echo "==== My ChIPSeq pipeline"
multirep_macs2_pipeline.pl \
--chip file1.bam,file2.bam \
--name chip1 \
--chip file3.bam,file4.bam \
--name chip2 \
--chip file5.bam,file6.bam \
--name chip3 \
--control input1.bam,input2.bam \
--out all_chip \
--dir ChIPSeq \
--pe \
--optdist 10000 \
--dupfrac 0.05 \
--cutoff 2 \
--peaksize 500 \
--peakgap 200 \
--species human \
--chrskip 'chrM' \
--blacklist /tomato/dev/data/Human/Hg38/hg38.blacklist.bed \
--plot \
--organize \
